  • IFieldView

2. ColumnSchema类型

  • 参看ColumnSchema
class ColumnSchema(string name, string sqlType = "INT") : IColumn {
	string SqlType { get; }
	string Default { get; set; }
	bool NotNull { get; set; }
	ColumnType ColumnType { get; set; }
}
